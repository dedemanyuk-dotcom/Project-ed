// Spreadsheet: 2 Sheet // Sheet "Master": barcode | nama | kategori | satuan | shelf_life_days // Sheet "Expired": barcode | nama | kategori | jumlah | satuan // | tgl_masuk | tgl_expired | lokasi | pic | catatan | created_at function doGet(e) { var a = e.parameter.action; var ss = SpreadsheetApp.getActiveSpreadsheet(); if (a === 'lookup') { var bc = e.parameter.barcode || ''; var d = ss.getSheetByName('Master').getDataRange().getValues(); for (var i = 1; i < d.length; i++) { if (String(d[i][0]).trim() === String(bc).trim()) { return ContentService.createTextOutput(JSON.stringify({ found: true, barcode: d[i][0], namaProduk: d[i][1], kategori: d[i][2] || '', satuan: d[i][3] || 'pcs', shelfLife: parseInt(d[i][4]) || 0 })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput( JSON.stringify({found:false}) ).setMimeType(ContentService.MimeType.JSON); } if (a === 'sync') { var d = ss.getSheetByName('Master').getDataRange().getValues(); var items = []; for (var i = 1; i < d.length; i++) { if (d[i][0] && d[i][1]) { items.push({ barcode: String(d[i][0]).trim(), namaProduk: d[i][1], kategori: d[i][2] || '', satuan: d[i][3] || 'pcs', shelfLife: parseInt(d[i][4]) || 0 }); } } return ContentService.createTextOutput( JSON.stringify({items:items, total:items.length}) ).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput('OK'); } function doPost(e) { var d = JSON.parse(e.postData.contents); ss.getSheetByName('Expired').appendRow([ d.barcode, d.namaProduk, d.kategori, d.jumlah, d.satuan, d.tanggalMasuk, d.tanggalExpired, d.lokasi, d.pic || '', d.catatan || '', new Date().toLocaleString('id-ID') ]); return ContentService.createTextOutput( JSON.stringify({status:'ok'}) ).setMimeType(ContentService.MimeType.JSON); } // Deploy > New deployment > Web app // Execute as: Me | Who has access: Anyone
